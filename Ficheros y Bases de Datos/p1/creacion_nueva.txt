CREATE TABLE Bibus(
matricula CHAR(8), 
estado VARCHAR2(20) NOT NULL,
ultima_ITV DATE NOT NULL,
siguiente_ITV DATE NOT NULL,
CONSTRAINT pk_Bibus
PRIMARY KEY (matricula),
CONSTRAINT ck1_bibus CHECK (siguiente_ITV >= ultima_ITV),
CONSTRAINT ck2_bibus CHECK (estado IN (‘asignado’, ‘revision’, ‘disponible’))
);


CREATE TABLE Bibusero(
pasaporte CHAR(20),
nombre VARCHAR2(80) NOT NULL,
apellido1 VARCHAR2(80) NOT NULL,
apellido2 VARCHAR2(80),
telefono NUMBER(9) NOT NULL,
email VARCHAR2(100) NOT NULL,
fecha_inicio DATE NOT NULL,
fecha_fin DATE,
estado VARCHAR2(20) NOT NULL,
CONSTRAINT pk_Bibusero 
PRIMARY KEY(pasaporte),
CONSTRAINT ck_contrato CHECK (fecha_inicio <= sysdate()),
CONSTRAINT ck_fecha_fin CHECK (fecha_fin IS NULL OR fecha_fin > fecha_inicio),
CONSTRAINT ck_bibusero CHECK (estado IN (‘asignado’, ‘libranza’, ‘sucursal’))
);






CREATE TABLE Municipio(
nombre VARCHAR2(50),
provincia VARCHAR2(22),
poblacion NUMBER(4) NOT NULL,
tiene_biblioteca CHAR(1) NOT NULL,
CONSTRAINT pk_Municipio 
PRIMARY KEY(nombre, provincia),
CONSTRAINT ck_tiene_biblioteca CHECK (tiene_biblioteca IN (‘Y’, ‘N’))
);


CREATE TABLE Autor (
nombre VARCHAR2(100),
apellido VARCHAR2(100),
fecha_nacimiento DATE,
fecha_defuncion DATE,
CONSTRAINT pk_autor 
PRIMARY KEY (nombre, apellido),
CONSTRAINT ck_fecha_nacimiento CHECK (fecha_nacimiento <= sysdate),
CONSTRAINT ck_fecha_defuncion CHECK (fecha_defuncion IS NULL OR (fecha_defuncion > fecha_nacimiento)AND fecha_defuncion <= sysdate)))
);




CREATE TABLE Lengua(
idioma VARCHAR2(50),
CONSTRAINT pk_Lengua
PRIMARY KEY (idioma)
);


CREATE TABLE Tema (
tema VARCHAR2(100),
CONSTRAINT pk_Tema
PRIMARY KEY (tema)
);


CREATE TABLE Ruta(
bibus_matricula CHAR(8),
fecha DATE,
bibusero_pasaporte CHAR(20) NOT NULL,
CONSTRAINT pk_ruta 
PRIMARY KEY(bibus_matricula, fecha),
CONSTRAINT u_ruta
UNIQUE(fecha, bibusero_pasaporte),
CONSTRAINT fk_Ruta_Bibus 
FOREIGN KEY(bibus_matricula) 
REFERENCES Bibus(matricula),
CONSTRAINT fk_Ruta_Bibusero
FOREIGN KEY (bibusero_pasaporte)
REFERENCES Bibusero(pasaporte),
CONSTRAINT ck_ruta CHECK (fecha >= sysdate)
);


CREATE TABLE Parada(
municipio_nombre VARCHAR2(50),
municipio_provincia VARCHAR2(22),
ruta_bibus_matricula CHAR(8),
ruta_fecha DATE,
tiempo TIMESTAMP NOT NULL,
CONSTRAINT pk_Parada
PRIMARY KEY (municipio_nombre, ruta_bibus_matricula, ruta_fecha),
CONSTRAINT fk_Parada_Municipio
FOREIGN KEY (municipio_nombre, municipio_provincia)
REFERENCES Municipio(nombre, provincia),
CONSTRAINT fk_Parada_Ruta
FOREIGN KEY (ruta_bibus_matricula, ruta_fecha)
REFERENCES Ruta(bibus_matricula, fecha)
);


CREATE TABLE Biblioteca(

fecha_fundacion DATE NOT NULL,
municipio_nombre VARCHAR2(50) NOT NULL,
municipio_provincia VARCHAR2(22) NOT NULL,
direccion VARCHAR2(100) NOT NULL,
email VARCHAR2(100) NOT NULL,
telefono NUMBER(9) NOT NULL,
CONSTRAINT pk_Biblioteca
PRIMARY KEY (cif),
CONSTRAINT fk_Biblioteca_Municipio
FOREIGN KEY (municipio_nombre, municipio_provincia)
REFERENCES Municipio(nombre, provincia),
CONSTRAINT ck_biblioteca CHECK (fecha_fundacion <= sysdate)


);




CREATE TABLE Libro(

autor1_nombre VARCHAR2(100),
autor1_apellido VARCHAR2(100),
pais VARCHAR2(50) NOT NULL,
fecha_publicacion DATE NOT NULL,
lengua VARCHAR2(50) NOT NULL,
titulo_alternativo VARCHAR2(200),
nota_contenido VARCHAR2(2500),
mencion_autores VARCHAR2(200),
CONSTRAINT pk_Libro
PRIMARY KEY(titulo, autor1_nombre, autor1_apellido),
CONSTRAINT fk_Libro_Autor
FOREIGN KEY(autor1_nombre, autor1_apellido)
REFERENCES Autor(nombre, apellido),

);

CREATE TABLE Edicion(
isbn NUMBER(13),
libro_titulo VARCHAR2(200) NOT NULL,
libro_autor1_nombre VARCHAR2(100) NOT NULL,
libro_autor1_apellido VARCHAR2(100) NOT NULL,
lengua_principal VARCHAR2(100) NOT NULL,
edicion NUMBER(2) NOT NULL,
editorial VARCHAR2(100) NOT NULL,
extension NUMBER(4) NOT NULL,
serie_nombre VARCHAR2(100), 
serie_numero NUMBER(5), 
deposito_legal VARCHAR2(12) NOT NULL,
lugar_publicacion VARCHAR2(50) NOT NULL,
dimension VARCHAR2(20) NOT NULL,
caracteristicas_fisicas VARCHAR2(50) NOT NULL,
notas VARCHAR2(500),
ID_BibliotecaNacional VARCHAR(200) NOT NULL,
URL VARCHAR2(200) NOT NULL,
CONSTRAINT pk_Edicion
PRIMARY KEY(isbn),
CONSTRAINT fk_Edicion_Libro
FOREIGN KEY(libro_titulo, libro_autor1_nombre, libro_autor1_apellido)
REFERENCES Libro(titulo, autor1_nombre, autor1_apellido),
CONSTRAINT fk_Edicion_Lengua
FOREIGN KEY(lengua_principal)
REFERENCES Lengua(idioma),
CONSTRAINT ck_numero_serie CHECK(serie_numero IS NULL OR serie_numero >= 1),
CONSTRAINT ck_edicion CHECK(edicion IS NULL OR edicion >= 1)
);


CREATE TABLE Ejemplar(

edicion_isbn NUMBER(13) NOT NULL,
biblioteca_cif CHAR(20),
bibus_matricula CHAR(8),
CONSTRAINT pk_Ejemplar
PRIMARY KEY(signatura),
CONSTRAINT fk_Ejemplar_Edicion
FOREIGN KEY(edicion_isbn)
REFERENCES Edicion(isbn),
CONSTRAINT fk_Ejemplar_Biblioteca
FOREIGN KEY(biblioteca_cif)
REFERENCES Biblioteca(cif),
CONSTRAINT fk_Ejemplar_Bibus
FOREIGN KEY(bibus_matricula)
REFERENCES Bibus(matricula),
CONSTRAINT ck_ejemplar CHECK (estado IN (‘nuevo’, ‘bueno’, ‘gastado’, ‘muy usado’, ‘deteriorado’))
);

CREATE TABLE Prestamo(

fecha_fin DATE NOT NULL,
CONSTRAINT pk_Prestamo
